123 private int loadDataFromDirectory(String theDirectoryName) throws IOException {
124 int count = 0;
125 ourLog.info("Reading files in directory: {}", theDirectoryName);
126 ClassPathResource dir = new ClassPathResource(theDirectoryName);
127 Collection<File> files = FileUtils.listFiles(dir.getFile(), null, false);
128 ourLog.info("{} files found.", files.size());
129 for (File file : files) {
130 String filename = file.getAbsolutePath();
131 ourLog.info("Processing filename '{}'", filename);
132 if (filename.endsWith(".cql") || filename.contains("expectedresults")) {
133 // Ignore .cql and expectedresults files
134 ourLog.info("Ignoring file: '{}'", filename);
135 } else if (filename.endsWith(".json")) {
136 if (filename.contains("bundle")) {
137 loadBundle(filename, ourCtx, ourClient);
138 } else {
139 loadResource(filename, ourCtx, myDaoRegistry);
140 }
141 count++;
142 } else {
143 ourLog.info("Ignoring file: '{}'", filename);
144 }
145 }
146 return count;
147 }

149 private Bundle loadBundle(String theLocation, FhirContext theCtx, IGenericClient theClient) throws IOException {
150 String json = stringFromResource(theLocation);
151 Bundle bundle = (Bundle) theCtx.newJsonParser().parseResource(json);
152 Bundle result = theClient.transaction().withBundle(bundle).execute();
153 return result;

159 * Create subscription

161 Subscription subscription = new Subscription();
162 subscription.setReason("Monitor new neonatal function (note, age will be determined by the monitor)");
163 subscription.setStatus(Subscription.SubscriptionStatus.REQUESTED);
164 subscription.setCriteria("Observation?status=final");

166 Subscription.SubscriptionChannelComponent channel = new Subscription.SubscriptionChannelComponent();
167 channel.setType(Subscription.SubscriptionChannelType.WEBSOCKET);
168 channel.setPayload("application/json");
169 subscription.setChannel(channel);

171 MethodOutcome methodOutcome = ourClient.create().resource(subscription).execute();
172 IIdType mySubscriptionId = methodOutcome.getId();
173
174 // Wait for the subscription to be activated
175 waitForSize(1, () -> ourClient.search().forResource(Subscription.class).where(Subscription.STATUS.exactly().code("active")).cacheControl(new CacheControlDirective().setNoCache(true)).returnBundle(Bundle.class).execute().getEntry().size());
176
177 /*
178 * Attach websocket
179 */
180
181 WebSocketClient myWebSocketClient = new WebSocketClient();
182 SocketImplementation mySocketImplementation = new SocketImplementation(mySubscriptionId.getIdPart(), EncodingEnum.JSON);
183
184 myWebSocketClient.start();
185 URI echoUri = new URI("ws://localhost:" + port + "/websocket");
186 ClientUpgradeRequest request = new ClientUpgradeRequest();
187 ourLog.info("Connecting to : {}", echoUri);
188 Future<Session> connection = myWebSocketClient.connect(mySocketImplementation, echoUri, request);
189 Session session = connection.get(2, TimeUnit.SECONDS);
190
191 ourLog.info("Connected to WS: {}", session.isOpen());
192
193 /*
194 * Create a matching resource
195 */
196 Observation obs = new Observation();
197 obs.setStatus(Observation.ObservationStatus.FINAL);
198 ourClient.create().resource(obs).execute();
199
200 // Give some time for the subscription to deliver

202
