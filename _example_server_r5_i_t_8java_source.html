83 subscription.setStatus(Enumerations.SubscriptionState.REQUESTED);
84 subscription.getChannelType()

89 MethodOutcome methodOutcome = ourClient.create().resource(subscription).execute();
90 IIdType mySubscriptionId = methodOutcome.getId();
91
92 // Wait for the subscription to be activated
93 waitForSize(1, () -> ourClient.search().forResource(Subscription.class).where(Subscription.STATUS.exactly().code("active")).cacheControl(new CacheControlDirective().setNoCache(true)).returnBundle(Bundle.class).execute().getEntry().size());
94
95 /*
96 * Attach websocket
97 */
98
99 WebSocketClient myWebSocketClient = new WebSocketClient();
100 SocketImplementation mySocketImplementation = new SocketImplementation(mySubscriptionId.getIdPart(), EncodingEnum.JSON);
101
102 myWebSocketClient.start();
103 URI echoUri = new URI("ws://localhost:" + port + "/websocket");
104 ClientUpgradeRequest request = new ClientUpgradeRequest();
105 ourLog.info("Connecting to : {}", echoUri);
106 Future<Session> connection = myWebSocketClient.connect(mySocketImplementation, echoUri, request);
107 Session session = connection.get(2, TimeUnit.SECONDS);
108
109 ourLog.info("Connected to WS: {}", session.isOpen());
110
111 /*
112 * Create a matching resource

114 Observation obs = new Observation();
115 obs.setStatus(Enumerations.ObservationStatus.FINAL);
116 ourClient.create().resource(obs).execute();
117
118 /*

121 await().until(() -> mySocketImplementation.myPingCount > 0);
