155 public static Map<String, Object> getAllProperties(PropertySource<?> aPropSource) {
156 Map<String, Object> result = new HashMap<>();
157
158 if (aPropSource instanceof CompositePropertySource) {
159 CompositePropertySource cps = (CompositePropertySource) aPropSource;
160 cps.getPropertySources().forEach(ps -> addAll(result, getAllProperties(ps)));
161 return result;
162 }
163
164 if (aPropSource instanceof EnumerablePropertySource<?>) {
165 EnumerablePropertySource<?> ps = (EnumerablePropertySource<?>) aPropSource;
166 Arrays.asList(ps.getPropertyNames()).forEach(key -> result.put(key, ps.getProperty(key)));
167 return result;
168 }
169
170 return result;
171
172 }

174 private static void addAll(Map<String, Object> aBase, Map<String, Object> aToBeAdded) {

181 }
